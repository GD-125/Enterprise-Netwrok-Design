Configure ISP Router

enable
configure terminal

hostname ISP

interface GigabitEthernet0/0
 description TO_CLOUD
 ip address 200.1.1.1 255.255.255.252
 no shutdown
 exit

interface GigabitEthernet0/1
 description TO_FIREWALL
 ip address 100.100.100.1 255.255.255.252
 no shutdown
 exit

end
write

Configure Static Routes
ISP Router
The ISP Router only knows the directly connected networks. You must tell it how to reach all your internal VLANs.

enable
configure terminal
ip route 192.168.10.0 255.255.255.0 100.100.100.2
ip route 192.168.20.0 255.255.255.0 100.100.100.2
ip route 192.168.30.0 255.255.255.0 100.100.100.2
ip route 192.168.40.0 255.255.255.0 100.100.100.2
ip route 192.168.50.0 255.255.255.0 100.100.100.2
ip route 192.168.99.0 255.255.255.0 100.100.100.2
end
write
